주어진 설명을 따라서 출력하면 된다

import sys

def solution():
    # 받을 값을 integer 타입의 a, b, c로 선언한다
    a, b, c = map(int, sys.stdin.readline().strip().split(" "))

    print((a+b)%c)
    print(((a%c) + (b%c))%c)
    print((a*b)%c)
    print(((a%c) * (b%c))%c)
    
if __name__ == "__main__":
    solution()
